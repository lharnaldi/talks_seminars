\begin{frame}
	\frametitle{Laboratorio electrónico open source}
	\begin{block}{Características}
		\begin{itemize}
			\item Herramienta para medir y adquirir datos. 
			\item Está hecho en torno a un SoC {\color{blue}Zynq 7010} de Xilinx, que
						combina un procesador ARM dual-core y una FPGA.
			\item Fundado  a través de una campaña 
						{\href{http://www.open-electronics.org/open-instruments-for-everyone-by-red-pitaya-is-on-kickstarter}{KickStarter}}
						en 2013. 
				\begin{itemize}
					\item {\alert{La idea requería fondos por U\$D 50.000, pero obtuvo U\$D
								256.000 de sus seguidores.}} 
				\end{itemize}
			\item Puede albergar una distribución GNU/Linux para el manejo completo de
						la tarjeta.
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Laboratorio electrónico open source}
	\begin{block}{Características (Cont.)}
		\begin{itemize}
			\item Tamaño reducido (tamaño similar a una tarjeta de crédito - no a la
						cuenta de esa tarjeta ;) ).
			\item Ancho de banda de trabajo razonable (50\,MHz).
			\item Permite trabajar con código de bajo y alto nivel: Python, C/C++,
						Matlab y LabView.
			\item \color{blue}{Extremadamente flexible para el desarrollo de aplicaciones ad-hoc,
						\alert{LAGO} por ejemplo :)}
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}

		\begin{exampleblock}{Características}
			\begin{itemize}
				\item Basada en cálculos con FFT
				\item Ancho de banda 0 - 62\,MHz 
				\item Rango dinámico - 80\,dBm
				\item Dos canales de entrada (conectores SMA)
				\item Nivel de ruido de entrada < -119\,dBm/Hz
				\item Resolución de 14 bit
				\item Rango de entrada +/- 1 V (LV) LV- low voltage jumper settings

				\item Acoplamiento de entrada DC
