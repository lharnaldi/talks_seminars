\section{Formato de trabajo}
%------------------------------------------------------------------------------
\begin{frame}
	\begin{center}
		\Huge{\color{blue}{Formato de trabajo experimental}}
	\end{center}
\end{frame}

\begin{frame}
	\begin{block}{¿Debemos entregar el experimento completamente armado?}
  	\begin{itemize}
  		\item	Siempre se debe dejar un poco de elaboración al alumno
  		\item	Nosotros debemos ser sólo guías
				\begin{itemize}
					\item Dar un introducción teórica y dejar que (la mayoría de) los
problemas los resuelvan los alumnos. Cuando hablamos de problemas nos referimos
a cosas tan simples como la interconexión de los bloques del experimento de que
se trate.
					\item \alert{Esto contribuye a la formación de un espíritu crítico
científico}
				\end{itemize}
  	\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\begin{block}{¿Debemos entregar el experimento completamente armado? (Cont.)}
  	\begin{itemize}
  		\item	Hay dos opciones: experimentos completamente armados (ejemplo:
dispositivos PASCO) y elaboración propia de experimentos (ejemplo: Millikan con
elementos ``hecho en casa'')
  		\item	No hay \alert{uno mejor que el otro}, siempre que los lineamientos para
realizar la experiencia estén claros para los estudiantes en ambos casos.
  		\item	Debemos contar, siempre que sea posible, con las últimas
herramientas tanto de hardware como de software para la realización de los
experimentos.
  	\end{itemize}
	\end{block}
\end{frame}

\section{Formato de los informes experimentales}
%------------------------------------------------------------------------------
\begin{frame}
	\begin{center}
		\Huge{\color{blue}{Formato de los informes experimentales}}
	\end{center}
\end{frame}

\begin{frame}
  \frametitle{}
	\begin{center}
		\includegraphics[height=0.8\textheight,width=0.75\textwidth]{cnea_ib_lago/alumnos_exp}
	\end{center}
\end{frame}

\begin{frame}
			\begin{block}{¿Qué deben contener los informes?}
    		\begin{itemize}
      		\item	Estructurado o no estructurado
      		\item	¿Nosotros imponemos la forma del informe?
      		\item	Hay que tener en cuenta que: \alert{parte de la elaboración y
								asimilación de los conocimientos del experimento está en saber
								mostrarlo/explicarlo al resto del mundo}  
      		\item	Todos los puntos en cuanto a la elaboración del informe deben
								estar bien claros para el estudiante antes de iniciar la experiencia
    		\end{itemize}
			\end{block}
\end{frame}

\begin{frame}
	\frametitle{Partes del informe experimental}
	\begin{block}{}
		\begin{itemize}
			\item \alert{Título}
			\item \alert{Resumen}
			\item \alert{Introducción} 
				\begin{itemize}
					\item Pone en contexto el trabajo realizado
				\end{itemize}
			\item \alert{Desarrollo experimental}
				\begin{itemize}
					\item Explicación de cómo midieron lo que midieron
					\item Muestran cómo es la disposición de los aparator e instrumentos
								que utilizaron duranto el transcurso del experimento
				\end{itemize}
			\item \alert{Análisis de resultados}
				\begin{itemize}
					\item Es el tratamiento propiamente dicho de los datos obtenidos
					\item Aquí es donde el análisis y tratamiento de errores juega un papel
								fundamental
					\item Aquí es donde nosotros, como docentes, vemos si los
								conocimientos fueron adquiridos o no, ya que esta sección es de elaboración pura
								del estudiante
				\end{itemize}
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Partes del informe experimental (Cont.)}
	\begin{block}{}
		\begin{itemize}
			\item \alert{Conclusiones}
				\begin{itemize}
					\item Puede incluir recomendaciones de posibles mejoras al experimento
				\end{itemize}

			\item \alert{Apéndices}
				\begin{itemize}
					\item Se incluye todo lo que es secundario, o que por su extensión
								puede interferir con la explicación normal del experimento y sus resultados
					\item Como ejemplo puede ser el desarrollo de una fórmula para saber
								de dónde salió
				\end{itemize}
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Conclusiones}
	\begin{alertblock}{}
		\begin{itemize}
			\item Se vieron alguno de los factores que influyen en la sólida formación
de (estudiantes) investigadores
			\item Repasamos el software que se utiliza actualmente, no solo en las
universidades, sino también en la industria
			\item Vimos las opciones que tenemos a la hora de realizar los
experimentos y presentar sus resultados en un ambiente académico 
		\end{itemize}
	\end{alertblock}
