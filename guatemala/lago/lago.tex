\begin{frame}
  \frametitle{{\color{red}L}atin {\color{red}A}merican {\color{red}G}iant {\color{red}O}bservatory (LAGO)}
	\framesubtitle{Un ``arreglo'' muy grande de detectores de agua Cherenkov (WCD)}
			\begin{alertblock}{La Colaboración LAGO}
			\begin{itemize}
				\item Mas de 80 miembros de 25 instituciones en 9 países latinoamericanos.
				\item \textbf{Objetivos científicos:}\\
					\begin{itemize}
						\item Astropartículas hasta la ``\textit{rodilla}'' de los rayos cósmicos ($\sim10^{15}$ -- $10^{16}$ eVs)
						\item Estudiar transitorios y fenómenos de Space Weather de larga duración debidos a la modulación solar de los rayos cósmicos.
						\item Medición de la radiación de fondo a nivel de tierra
					\end{itemize}
				\item \textbf{Objetivos académicos:}\\
					\begin{itemize}
						\item Entrenar a estudiantes latinoamericanos en técnicas de física de altas energías, sofware (análisis de datos) y técnicas de astropartículas.
						\item Construir una red latinoamericana de investigadores en astropartículas.
					\end{itemize}
			\end{itemize}
		\end{alertblock}	
\end{frame}

\begin{frame}
\frametitle{Nuestro detector: WCD (Water Cherenkov Detector)}
	\begin{columns}
		\begin{column}{0.55\textwidth}
			\begin{block}{}
    		\begin{itemize}
      		\item	Detector autónomo, simple, confiable y barato.
      		\item	Tanques comerciales con 1.5 $m^2$ -- 10 $m^2$ de área de
detección, llenos de agua purificada.
      		\item	Covertura interior de Tyvek (difusivo y dispersivo)
					\item PMT + tarjeta de digitalización (diseño propio) 
					\item FPGA + Raspberry Pi: control del detector, telemetría,
adquisición de datos y pre-análisis de los datos on-board.
					\item Sincronización temporal: GPS con señal PPS 
					\item Consumo de la estación $\sim$ 8 W.
    		\end{itemize}
			\end{block}
		\end{column} \ \
		\begin{column}{0.45\textwidth}
  		\includegraphics[width=\textwidth]{lago/lago_daq}
		\end{column}
	\end{columns}
